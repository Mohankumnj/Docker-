pipeline {
    agent any
    
    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }
        stage('Build and Push Docker Image') {
            steps {
                script {
                    def imageName = "my-docker-image:latest"
                    sh "docker build -t $imageName ."
                    sh "docker push $imageName"
                }
            }
        }
        stage('Deploy') {
            steps {
                sh "docker run -d -p 8080:80 my-docker-image:latest"
            }
        }
    }
}
